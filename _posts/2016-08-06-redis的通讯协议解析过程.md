# redis的通讯协议解析过程

源码来自redis2.8.24

* redis客户端与服务器通信时传递的是什么格式的数据？
* 我能自己写一个redis的客户端吗？
* 为什么redis是二进制安全的？


首先看看redis的基本交互协议：

```
SET命令:
    set name joel
	客户端实际传递给服务器的数据如下：
        *3\r\n         		//参数的数量
	$3\r\n			//第一个参数“set”的长度
        set\r\n			//第一个参数
        $4\r\n			//第二个参数的长度
        name\r\n		//第二个参数：key
	$4\r\n			//第三个参数的长度
	joel\r\n		//第三个参数：value

    服务端操作成功返回数据：
        +OK\r\n

    如果出现错误服务端会返回：
        -错误信息\r\n

```

在客户端与服务器建立连接之后，会触发服务器端的io事件的回调，将客户端传来的数据读取存放在client->querybuf中，

随后就会调用processMultibulkBuffer函数，解析保存在querybuf中的数据，并将解析到的参数填充在client->argv中。

```c
//在networking.c的927行
int processMultibulkBuffer(redisClient *c) {
    char *newline = NULL;
    int pos = 0, ok;
    long long ll;
    //一般命令c->multibulklen开始就是0
    if (c->multibulklen == 0) {
        /* The client should have been reset */
        redisAssertWithInfo(c,NULL,c->argc == 0);

        /* Multi bulk length cannot be read without a \r\n */
        newline = strchr(c->querybuf,'\r'); //定位到第一个换行符的位置
        if (newline == NULL) {
            if (sdslen(c->querybuf) > REDIS_INLINE_MAX_SIZE) {
                addReplyError(c,"Protocol error: too big mbulk count string");
                setProtocolError(c,0);
            }
            return REDIS_ERR;
        }

        /* Buffer should also contain \n */
        //通信协议中的换行符其实是\r\n,所以有－2这么一说，这一步也是验证换行符是否正确 
        if (newline-(c->querybuf) > ((signed)sdslen(c->querybuf)-2))
            return REDIS_ERR;

        /* We know for sure there is a whole line since newline != NULL,
         * so go ahead and find out the multi bulk length. 
         * /
         //命令的参数必须以＊开头
        redisAssertWithInfo(c,NULL,c->querybuf[0] == '*');
        //获取“参数的数量”，检查其长度是否超标
        //querybuf+1是为了跨过起始的‘＊’
        //string2ll函数是string转long long的操作
        ok = string2ll(c->querybuf+1,newline-(c->querybuf+1),&ll);
        if (!ok || ll > 1024*1024) {
            addReplyError(c,"Protocol error: invalid multibulk length");
            setProtocolError(c,pos);
            return REDIS_ERR;
        }
		
	/* newline是当前第一个换行符所在的位置，
	 * 它减去首指针位置得到第一段参数的长度，也就是换行符所在的位置
	 * 再加上2，则是跨过换行符,跨过这一整行
	 */
        pos = (newline-c->querybuf)+2;
        if (ll <= 0) { //普通命令不会执行到这
            sdsrange(c->querybuf,pos,-1); //把第二行的内容移到第一行，覆盖第一行
            return REDIS_OK;
        }

        c->multibulklen = ll; //保存“参数的数量”

        /* Setup argv array on client structure */
        if (c->argv) zfree(c->argv);
        //为参数们申请指针空间，可以看出参数会用robj来封装
        c->argv = zmalloc(sizeof(robj*)*c->multibulklen); 
    }
	 
    //参数长度不得少于1, redisAssertWithInfo验证不通过,程序直接退出exit(1)
    redisAssertWithInfo(c,NULL,c->multibulklen > 0);
    while(c->multibulklen) {
        /* Read bulk length if unknown */
        if (c->bulklen == -1) {
            //跨过上一行，定位到下一行
            newline = strchr(c->querybuf+pos,'\r');
            if (newline == NULL) {
                if (sdslen(c->querybuf) > REDIS_INLINE_MAX_SIZE) {
                    addReplyError(c,
                        "Protocol error: too big bulk count string");
                    setProtocolError(c,0);
                    return REDIS_ERR;
                }
                break;
            }

            /* Buffer should also contain \n */
            //判断换行符的位置是否超过了参数字符串最后的换行符之前的位置
            if (newline-(c->querybuf) > ((signed)sdslen(c->querybuf)-2))
                break;
	   /* 参数格式都是:
	    * 	$3\r\n //参数长度
	    * 	set //参数
	    * 	...
	    * 现在开始读取标注长度的那一行
	    * 判断参数是否以’$‘开头
	    */
            if (c->querybuf[pos] != '$') {
                addReplyErrorFormat(c,
                    "Protocol error: expected '$', got '%c'",
                    c->querybuf[pos]);
                setProtocolError(c,pos);
                return REDIS_ERR;
            }
			  
	    //获取参数的长度
            ok = string2ll(c->querybuf+pos+1,newline-(c->querybuf+pos+1),&ll);
            //验证值长度是否超过512M
            if (!ok || ll < 0 || ll > 512*1024*1024) {
                addReplyError(c,"Protocol error: invalid bulk length");
                setProtocolError(c,pos);
                return REDIS_ERR;
            }
				
	    //跨过当前解析的“参数长度”这一行
            pos += newline-(c->querybuf+pos)+2;
            //如果接下来的数据是大块的数据，那么就要特殊处理一下
            if (ll >= REDIS_MBULK_BIG_ARG) {
                size_t qblen;

                /* If we are going to read a large object from network
                 * try to make it likely that it will start at c->querybuf
                 * boundary so that we can optimize object creation
                 * avoiding a large copy of data. */
                sdsrange(c->querybuf,pos,-1);
                pos = 0;
                qblen = sdslen(c->querybuf);
                /* Hint the sds library about the amount of bytes this string is
                 * going to contain. */
                if (qblen < (size_t)ll+2)
                    c->querybuf = sdsMakeRoomFor(c->querybuf,ll+2-qblen);
            }
            c->bulklen = ll;
        }
			
	//开始读取命令字符串
        /* Read bulk argument */
        //判断下一行的位置是否超过了参数字符串的最大长度
        if (sdslen(c->querybuf)-pos < (unsigned)(c->bulklen+2)) {
            /* Not enough data (+2 == trailing \r\n) */
            break;
        } else {
            /* Optimization: if the buffer contains JUST our bulk element
             * instead of creating a new object by *copying* the sds we
             * just use the current sds string. */
            if (pos == 0 &&
                c->bulklen >= REDIS_MBULK_BIG_ARG &&
                (signed) sdslen(c->querybuf) == c->bulklen+2)
            {
            	   //普通的命令不会触发这部分
                c->argv[c->argc++] = createObject(REDIS_STRING,c->querybuf);
                sdsIncrLen(c->querybuf,-2); /* remove CRLF */
                c->querybuf = sdsempty();
                /* Assume that if we saw a fat argument we'll see another one
                 * likely... */
                c->querybuf = sdsMakeRoomFor(c->querybuf,c->bulklen+2);
                pos = 0;
            } else {
            	/* 解析提取参数
            	 * c->querybuf+pos 是当前参数的字符串起始地址，
            	 * c->bulklen就是协议中表明的参数长度
            	 * 按照参数中指定的长度来读取数据，这就是redis实现二进制安全的手段之一
            	 */
                c->argv[c->argc++] =
                    createStringObject(c->querybuf+pos,c->bulklen); 
                pos += c->bulklen+2; //pos跨过当前行的换行符
            }
            c->bulklen = -1;
            c->multibulklen--;
        }
    }
	
    //当最后一个参数都被读了，那么就再位移一次，把所有的值都覆盖掉。
    //注意，虽然覆盖掉了，但并没有内存回收操作！
    /* Trim to pos */
    if (pos) sdsrange(c->querybuf,pos,-1);

    /* We're done when c->multibulk == 0 */
    //正常结束的情况是读完了请求中所标注的所有的参数
    if (c->multibulklen == 0) return REDIS_OK;

    /* Still not read to process the command */
    return REDIS_ERR;
}


```

#### 为什么redis中每个参数字段都需要显示的标注长度？按照参数中指定的长度来读取数据，为什么会是redis实现二进制安全的手段之一？到底什么是二进制安全？
现在我们先假设redis通信协议中参数不标明长度，如果传递的数据只是字符串（key是字符串，value也是字符串），那么我们仍旧能按照换行符分割，一行一行读取；

但如果我们传递的是二进制的数据，那就不能按照普通字符串的方式来操作，例如无符号数字65791，它的二进制编码为“0*100ff”，数据中间的一个字节是0，字符串的结束标记也是｀\0｀,那么这个“0字节”在普通的字符串处理函数看来会认为是个字符串结束标记，所以“0字节”之后的数据就被丢弃。

这种字符串的处理方式就不是二进制安全的，所以最简单的方法就是采用标注长度的方式，数据的接收方要分割读取数据，但不知道每一段在什么地方结束，所以只能是你让我读多少我就读多少，这就是二进制安全的。
